1   package pages;
2   
3   import io.restassured.RestAssured;
4   import io.restassured.response.Response;
5   import org.json.JSONObject;
6   
7   import static io.restassured.RestAssured.given;
8   
9   public class BookingPage {
10  
11      public BookingPage() {
12          RestAssured.baseURI = "https://restful-booker.herokuapp.com";
13      }
14  
15      // Method to create a booking
16      public Response createBooking(JSONObject bookingPayload) {
17          return given()
18                  .header("Content-Type", "application/json")
19                  .body(bookingPayload.toString())
20                  .when()
21                  .post("/booking")
22                  .then()
23                  .extract().response();
24      }
25  
26      // Method to get a booking by ID
27      public Response getBookingById(int bookingId) {
28          return given()
29                  .header("Content-Type", "application/json")
30                  .when()
31                  .get("/booking/" + bookingId)
32                  .then()
33                  .extract().response();
34      }
35  
36      // Method to create a sample booking payload for the tests
37      public JSONObject createBookingPayload(String firstName, String lastName, double totalPrice, boolean depositPaid, String checkInDate, String checkOutDate, String additionalNeeds) {
38          JSONObject bookingPayload = new JSONObject();
39          bookingPayload.put("firstname", firstName);
40          bookingPayload.put("lastname", lastName);
41          bookingPayload.put("totalprice", totalPrice);
42          bookingPayload.put("depositpaid", depositPaid);
43  
44          // Add booking dates
45          JSONObject bookingDates = new JSONObject();
46          bookingDates.put("checkin", checkInDate);
47          bookingDates.put("checkout", checkOutDate);
48          bookingPayload.put("bookingdates", bookingDates);
49          bookingPayload.put("additionalneeds", additionalNeeds);
50  
51          return bookingPayload;
52      }
53  }
